#include<iostream>
using namespace std;
int main()

{
	int n ;
	int k = 1 ;

	int stacks[50] ;

	while ( cin >>n && n > 0 )

	{
		int counter1 = 0 ;
		int counter2 = 0 ;
		int x = n ; 
		int y ;

		for ( int i = 0 ; n > 0 ;i++ )

		{
			cin>>stacks[i];
			counter1+=stacks[i];

			n--;
		}


		y = counter1/x ;

		

		for ( int j = 0 ; j < x ; j++ )
			
		{
			    while ( stacks[j] >  y )  

				{

					counter2++;

					stacks[j]--;
				}




		}



		cout<<"Set #"<<k<<endl;
		cout<<"The minimum number of moves is "<<counter2<<"."<<endl;
		k++;

		cout<<endl;
	}

	return 0 ;
}
