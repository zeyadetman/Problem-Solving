#include<iostream>
#include<vector>
#include<algorithm>

using namespace std ;

int main()

{
	int n , m ;

	vector<int> arr ;

	while ( cin >> n && n > 0 )

	{
		for ( int i = 0 ; i < n ; i++ )

		{
			cin >> m ;

			arr.push_back(m);

		}

		sort ( arr.begin() , arr.end() ) ;


		for ( int j = 0 ; j < n ; j++ )

		{

			cout<<arr[j] ;

			if ( j != n-1 )

			{

				cout<<" " ;
			}

		}

		cout<<endl;


		while ( ! arr.empty() )

		{

			arr.pop_back() ;

		}




	}

	return 0 ;

}
