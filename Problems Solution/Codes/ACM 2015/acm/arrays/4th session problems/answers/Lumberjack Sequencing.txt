#include<iostream>
using namespace std;
int main()

{

	int t , i ;
	int lengths[10];

	cin >> t ;

	for ( i = 0 ; i < t ; i++ )

{
	for ( int j = 0 ; j < 10 ; j++ )

	{

		cin >> lengths[j] ;
	}

	int x = 0 ;

	while ( lengths[x] > lengths[x+1] && x != 9 )

	{
		x++;
	}

	int y = 0 ;

	while ( lengths[y] < lengths[y+1] && y != 9 )
	{
		y++ ;
	}

	if ( i ==0 ) 
	{
	cout<<"Lumberjacks:"<<endl;
	}

	if ( x == 9 || y == 9 )

	{
		cout<<"Ordered"<<endl;

	}

	else
	{
		cout<<"Unordered"<<endl;
	}


}

return 0 ;

}
