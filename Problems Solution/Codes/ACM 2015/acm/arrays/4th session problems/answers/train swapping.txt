#include<iostream>
using namespace std;
int main()

{
	int t , i , l ;

	int length[50];

	cin >> t ;

	for ( i = 0 ; i < t ; i++ )

	
	{
		int counter = 0 ;

		cin >> l ;

		for ( int y = 0 ;y<l ; y++ )

		{

			cin>>length[y];

		}

		for ( int j = 1; j<l ; j++ )

		{
			
		

			for ( int x = 0 ;x<= l-2 ;  x++ )

			{
				
				if ( length[x] > length[x+1] )
				{
					counter++ ;
					swap(length[x],length[x+1]);

				}

				

             
			}

		}


		cout<<"Optimal train swapping takes "<<counter<<" swaps."<<endl;
	}

return 0 ;

}
