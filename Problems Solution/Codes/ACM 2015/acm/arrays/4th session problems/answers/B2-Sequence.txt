#include<iostream>
using namespace std;
int main()
{
 
	int n , i = 1 , c ;
	int numbers[99] ;
	int sums[10000] ;
	bool result  ;




	while ( cin >> n )

{
	result = true ;


	for ( int j = 0 ; j < n ; j++ )

{
	cin >> numbers[j] ;
}

	for ( int b = 0 ; b < n-1 ; b++ )

	{
		if (numbers[b+1] <= numbers[b] || numbers[b] < 1 )
		{
			result = false ;

			break;

		}
	}


	c = 0 ;

	for ( int x = 0 ; x < n ; x++ )

{
	for ( int y = x ; y < n ; y++ )
	{

		sums[c] = numbers[x] + numbers[y] ;

		c++ ;
	}

}
	for ( int z = 0 ; z < c ; z++ )

{
	for ( int a = 0 ; a < c ;a++)

	{
		if ( sums[z] == sums[a] && a != z)
		{
			result = false ;
		}

		else 

		{
			continue ;
		}

	}

}

	if ( result == true )

	{
		cout<<"Case #"<<i<<": "<<"It is a B2-Sequence."<<endl;
	}


	else if ( result == false )

	{
		cout<<"Case #"<<i<<": "<<"It is not a B2-Sequence."<<endl;
	}

	cout << endl;

	i++ ;


}
return 0 ;
}
